
SYSTEM INFO (1-15)

1. Get-WmiObject -Class Win32_OperatingSystem | Select Caption,Version,BuildNumber,OSArchitecture
2. Get-WmiObject -Class Win32_ComputerSystem | Select Name,Model,Manufacturer,TotalPhysicalMemory
3. Get-WmiObject -Class Win32_Processor | Select Name,NumberOfCores,NumberOfLogicalProcessors,MaxClockSpeed
4. Get-WmiObject -Class Win32_BIOS | Select Manufacturer,Name,SerialNumber,Version
5. Get-WmiObject -Class Win32_BaseBoard | Select Manufacturer,Product,SerialNumber
6. Get-WmiObject -Class Win32_ComputerSystemProduct | Select Name,UUID,IdentifyingNumber
7. Get-WmiObject -Class Win32_SystemEnclosure | Select Manufacturer,Type
8. Get-WmiObject -Class Win32_PhysicalMemory | Select Capacity,Speed,Manufacturer
9. Get-WmiObject -Class Win32_NetworkAdapterConfiguration | Where IPEnabled -eq $true | Select Description,IPAddress,DNSHostName
10. Get-WmiObject -Class Win32_TimeZone | Select Caption,DaylightName,StandardName
11. Get-WmiObject -Class Win32_StartupCommand | Select Name,Command,Location,User
12. Get-WmiObject -Class Win32_Environment | Where Name -like "*PATH*" | Select Name,VariableValue
13. Get-WmiObject -Class Win32_LogicalDisk | Select DeviceID,Size,FreeSpace,DriveType
14. Get-WmiObject -Class Win32_Volume | Select Name,Capacity,FreeSpace,DriveLetter
15. Get-WmiObject -Class Win32_SystemProcesses | Measure-Object -Property WorkingSetSize -Sum

SERVICE ENUM (36-50)
36. Get-WmiObject -Class Win32_Service | Select Name,State,StartMode,PathName
37. Get-WmiObject -Class Win32_Service | Where State -eq "Running" | Select Name,PathName
38. Get-WmiObject -Class Win32_Service | Where StartMode -eq "Auto" | Select Name,PathName
39. Get-WmiObject -Class Win32_Service | Select Name,PathName | Where PathName -notlike "C:\Windows\*" | Where PathName -notlike "*system32*"
40. Get-WmiObject -Class Win32_Service | Where PathName -like "*.exe*" | Select Name,PathName,State
41. Get-WmiObject -Class Win32_Service | Select Name,StartName,State
42. Get-WmiObject -Class Win32_Service | Where StartName -notlike "LocalSystem" | Select Name,StartName
43. Get-WmiObject -Class Win32_ServiceControl | Select Name,AcceptStop,AcceptPause
44. Get-WmiObject -Class Win32_BaseService | Select Name,ProcessId,State
45. Get-WmiObject -Class Win32_Service | Where State -eq "Stopped" | Select Name,StartMode
46. Get-WmiObject -Class Win32_Service | Select Name,PathName,ProcessId | Where ProcessId -eq 0
47. Get-WmiObject -Class Win32_Service | Where PathName -like "svchost*" | Select Name,ProcessId
48. Get-WmiObject -Class Win32_Service | Select Name,PathName | Where { $_.PathName -match "\\Program Files\\" }
49. Get-WmiObject -Class Win32_Service | Where StartMode -eq "Disabled" | Select Name,State
50. Get-WmiObject -Class Win32_Service | Measure-Object | Select Count
